    <button style="margin-top:8px" onclick="sendBack()">Отправить данные в бот и закрыть</button>

    function sendBack() {

      const payload = { ok: true, when: new Date().toISOString() };

      Telegram.WebApp.sendData(JSON.stringify(payload)); // вернётся боту

      Telegram.WebApp.close();

    }
